15 Metodo de K-means

k <- list()
for (i in 1:10) {
  k[[i]]=kmeans(iris4,i)
}

16 Graficar 3 cluster en base a K-means

17 Calcular el % de cambio en las distancias

betweens <- list()
for (i in 1:10) {
  betweens[[i]] = (k[[i]]$betweenss / k[[i]]$totss)*100
}
